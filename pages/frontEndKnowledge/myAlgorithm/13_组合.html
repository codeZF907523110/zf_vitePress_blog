力扣id=77 组合

题目：

给定两个整数 n 和 k，返回范围 [1, n] 中所有可能的 k 个数的组合。你可以按 任何顺序 返回答案。

示例：

输入：n = 4, k = 2

输出：

[[2,4],[3,4],[2,3],[1,2],[1,3],[1,4]]

思路：

  1. 回溯：
  2. 需要注意：当前下标i只关注i后面的元素，所以需要从num开始遍历
js
var combine = function (n, k) {
  const res = []

  const dfs = (path, num) => {
    if (path.length === k){
        res.push([...path])
        return
    }

    for (let i=num;i<=n;i++) {
        path.push(i)
        dfs(path, i+1)
        path.pop()
    }
  }

  dfs([], 1)
  return res
};
